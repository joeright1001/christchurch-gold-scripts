// turns the add cart icon into a link pointing to the cart page and inserts the cms id into the url with ?id=cmsId
document.addEventListener("DOMContentLoaded", function() {
    // Select all elements with the class 'add-cart-icon'
    const cartIcons = document.querySelectorAll('.add-cart-icon');

    cartIcons.forEach(function(icon) {
        
        // 1. Listen for the click
        icon.addEventListener('click', function(e) {
            
            // CRITICAL STEP: Stop the click from triggering the parent Link Block
            e.preventDefault(); 
            e.stopPropagation(); 

            // 2. Get the CMS data
            const cmsID = this.getAttribute('data-value-cms');
            
            // 3. Redirect to the cart with the data
            if (cmsID) {
                // This matches your requested format: /app/cart?data-value-cms=ID
                window.location.href = '/app/cart?data-value-cms=' + cmsID;
            }
        });
    });
});
</script>
